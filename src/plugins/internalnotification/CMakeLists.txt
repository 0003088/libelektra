include (LibAddMacros)

add_plugin (internalnotification
	SOURCES
		internalnotification.h
		internalnotification.c
	ADD_TEST
	COMPILE_DEFINITIONS
		PLUGIN_INTERNALNOTIFICATION_VERBOSE=${PLUGIN_INTERNALNOTIFICATION_VERBOSE}
	)

MARK_AS_ADVANCED(
	PLUGIN_INTERNALNOTIFICATION_VERBOSE_ENABLED
)

IF (NOT PLUGIN_INTERNALNOTIFICATION_VERBOSE)
	SET (PLUGIN_INTERNALNOTIFICATION_VERBOSE "0" CACHE STRING "Enable verbose output for internalnotification plugin")
ENDIF (NOT PLUGIN_INTERNALNOTIFICATION_VERBOSE)
